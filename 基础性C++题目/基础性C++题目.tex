\subsection{结构体拷贝}
给出以下代码，其行为是？


\begin{minted}[mathescape,	
    linenos,
    numbersep=5pt,
    gobble=2,
    frame=lines,
    framesep=2mm]{c++}

    #include <iostream>

    struct foo{
        char a[16];
    };
    
    int main(){
        foo a {"Hello, World!"};
        auto b = a;
        a.a[0] = '\0';
        std::cout << b.a;
    }
\end{minted}

\begin{itemize}
    \item [A]
          无法编译/未定义行为
    \item [B]
          可编译，保证无输出
    \item [C]
          可编译，保证输出"Hello, World!"
    \item [D]
          可编译，输出为实现定义
\end{itemize}

\begin{minted}[mathescape,	
    linenos,
    numbersep=5pt,
    gobble=2,
    frame=lines,
    framesep=2mm]{c++}

    #include <iostream>

    struct foo{
        char a[16];
    };

    struct Base { int x; };
    struct Derived : Base { int y; }
    
    int get_xy(Base * param){
        auto pd = ????_cast<Derived*>(param);
        return pd->x * pd->y;
    }
\end{minted}

\begin{itemize}
    \item [A]
        static\_cast
    \item [B]
